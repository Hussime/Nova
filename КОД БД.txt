USE ReHub;
GO

-- Таблица ПРЕПОДАВАТЕЛЬ
CREATE TABLE Преподаватель (
    М_преподавателя INT IDENTITY(1,1) PRIMARY KEY,
    ФИО NVARCHAR(100) NOT NULL,
    Кафедра NVARCHAR(100),
    Email NVARCHAR(100),
    Телефон NVARCHAR(20),
    Логин NVARCHAR(50) NOT NULL UNIQUE,
    Пароль NVARCHAR(255) NOT NULL
);

-- Таблица СТУДЕНТ
CREATE TABLE Студент (
    М_студента INT IDENTITY(1,1) PRIMARY KEY,
    ФИО NVARCHAR(100) NOT NULL,
    Группа NVARCHAR(50) NOT NULL,
    Email NVARCHAR(100),
    Телефон NVARCHAR(20),
    Логин NVARCHAR(50) NOT NULL UNIQUE,
    Пароль NVARCHAR(255) NOT NULL
);

-- Таблица ФАКУЛЬТАТИВ
CREATE TABLE Факультатив (
    М_факультатива INT IDENTITY(1,1) PRIMARY KEY,
    Название NVARCHAR(100) NOT NULL,
    Описание NVARCHAR(500),
    М_преподавателя INT NOT NULL,
    Макс_количество INT DEFAULT 30,
    Статус NVARCHAR(20) DEFAULT 'Активен',
    Дата_создания DATETIME DEFAULT GETDATE(),
    Дата_занятия DATE, -- Добавлено по диаграмме
    Время_занятия TIME, -- Добавлено по диаграмме
    FOREIGN KEY (М_преподавателя) REFERENCES Преподаватель(М_преподавателя)
);

-- Таблица ЗАЯВКА
CREATE TABLE Заявка (
    М_заявки INT IDENTITY(1,1) PRIMARY KEY,
    М_студента INT NOT NULL,
    М_факультатива INT NOT NULL,
    Дата_подачи DATETIME DEFAULT GETDATE(),
    Статус NVARCHAR(20) DEFAULT 'Ожидание',
    Проверяет INT NULL,
    FOREIGN KEY (М_студента) REFERENCES Студент(М_студента),
    FOREIGN KEY (М_факультатива) REFERENCES Факультатив(М_факультатива),
    FOREIGN KEY (Проверяет) REFERENCES Преподаватель(М_преподавателя)
);

-- Таблица АДМИНИСТРАТОР
CREATE TABLE Администратор (
    М_администратора INT IDENTITY(1,1) PRIMARY KEY,
    ФИО NVARCHAR(100) NOT NULL,
    Логин NVARCHAR(50) NOT NULL UNIQUE,
    Пароль NVARCHAR(255) NOT NULL
);
